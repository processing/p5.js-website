    function setup() {
      createCanvas(100, 100);

      background(200);

      // Calculate cos() and acos() values.
      let a = PI;
      let c = cos(a);
      let ac = acos(c);

      // Display the values.
      text(`${round(a, 3)}`, 35, 25);
      text(`${round(c, 3)}`, 35, 50);
      text(`${round(ac, 3)}`, 35, 75);
