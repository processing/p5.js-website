  <p>创建一个沉浸式的 3D 背景。</p>

  <p><code>panorama()</code> 将包含 360˚ 内容的图像，例如地图或 HDRIs，转换为沉浸式的 3D 背景，围绕在绘图周围。探索空间需要通过诸如<a href="/reference/p5/orbitControl">orbitControl()</a> 或 <a href="/reference/p5/camera">camera()</a> 等函数来改变相机的视角。</p>

example:
  - |-

    <div class="notest">
    <code>
    // 单击并拖动鼠标以从不同角度查看场景。

    let img;

    // 加载图像并创建一个 p5.Image 对象。
    function preload() {
      img = loadImage('/assets/outdoor_spheremap.jpg');
    }

    function setup() {
      createCanvas(100 ,100 ,WEBGL);

      describe('A sphere floating above a landscape. The surface of the sphere reflects the landscape. The full landscape is viewable in 3D as the user drags the mouse.');
    }

    function draw() {
      // 添加全景图。
      panorama(img);

      // 启用鼠标轨道控制。
      orbitControl();

      // 使用图像作为光源。
      imageLight(img);

      // 为球体添加样式。
      noStroke();
      specularMaterial(50);
      shininess(200);
      metalness(100);

      // 绘制球体。
      sphere(30);
    }
    </code>
    </div>
