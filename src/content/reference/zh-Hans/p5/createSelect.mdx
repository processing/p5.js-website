description: >
  <p>创建一个下拉菜单<code>&lt;select&gt;&lt;/select&gt;</code>元素。</p>

  <p>参数是可选的。如果传递了<code>true</code>，如<code>let mySelect = createSelect(true)</code>，则下拉菜单将支持多选。如果传递了一个现有的<code>&lt;select&gt;&lt;/select&gt;</code>元素，如<code>let mySelect = createSelect(otherSelect)</code>，则现有元素将被包装在一个新的<a href="#/p5.Element">p5.Element</a>对象中。</p>

  <p>下拉菜单扩展了<a href="#/p5.Element">p5.Element</a>类，增加了一些管理选项的有用方法：</p>

  <ul>

  <li><code>mySelect.option(name, [value])</code>向菜单添加一个选项。第一个参数，<code>name</code>，是一个字符串，设置选项的名称和值。第二个参数，<code>value</code>，是可选的。如果提供了，它将设置对应于键<code>name</code>的值。如果一个选项的<code>name</code>已经存在，其值将被更改为<code>value</code>。</li>

  <li><code>mySelect.value()</code>返回当前选中选项的值。</li>

  <li><code>mySelect.selected()</code>返回当前选中的选项。</li>

  <li><code>mySelect.selected(option)</code>默认选择给定的选项。</li>

  <li><code>mySelect.disable()</code>将整个下拉菜单元素标记为禁用。</li>

  <li><code>mySelect.disable(option)</code>将给定的选项标记为禁用。</li>

  <li><code>mySelect.enable()</code>将整个下拉菜单元素标记为启用。</li>

  <li><code>mySelect.enable(option)</code>将给定的选项标记为启用。</li>

  </ul>

    // Hold CTRL to select multiple options on Windows and Linux.
    // Hold CMD to select multiple options on macOS.
    let mySelect;

    function setup() {
      createCanvas(100, 100);

      // Create a dropdown and allow multiple selections.
      // Place it beneath the canvas.
      mySelect = createSelect(true);
      mySelect.position(0, 100);

      // Add color options.
      mySelect.option('red');
      mySelect.option('green');
      mySelect.option('blue');
      mySelect.option('yellow');

      describe('A gray square with a dropdown menu beneath it. Colorful circles appear when their color is selected.');
    }

    function draw() {
      background(200);

      // Use the selected value(s) to draw circles.
      let colors = mySelect.selected();
      for (let i = 0; i < colors.length; i += 1) {
        // Calculate the x-coordinate.
        let x = 10 + i * 20;

        // Access the color.
        let c = colors[i];

        // Draw the circle.
        fill(c);
        circle(x, 50, 20);
      }
    }
