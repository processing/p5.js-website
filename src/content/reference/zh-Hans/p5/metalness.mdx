description: >
  <p>设置 <a href="/reference/p5/specularMaterial">specularMaterial()</a> 的"金属度"。</p>

  <p><code>metalness()</code> 可以使材料看起来更像金属。它影响材料反射光源的方式，包括 <a href="/reference/p5/directionalLight">directionalLight()</a>、<a href="/reference/p5/pointLight">pointLight()</a>、<a href="/reference/p5/spotLight">spotLight()</a> 和 <a href="/reference/p5/imageLight">imageLight()</a>。</p>

  <p>参数 <code>metallic</code> 是一个数字，表示金属度的程度。 <code>metallic</code> 必须大于 1 ，这是它的默认值。较高的值，比如 <code>metalness(100)</code>，会使高光材料看起来更像金属。</p>

    <code>
    function setup() {
      createCanvas(100, 100, WEBGL);

      describe(
        'Two blue spheres drawn on a gray background. White light reflects from their surfaces as the mouse moves. The right sphere is more metallic than the left sphere.'
      );
    }

    function draw() {
      background(200);

      // 打开一个环境光。
      ambientLight(200);

      // 获取鼠标的坐标。
      let mx = mouseX - 50;
      let my = mouseY - 50;

      // 打开一个跟随鼠标移动的白色点光源。
      pointLight(255, 255, 255, mx, my, 50);

      // 设定球体样式。
      noStroke();
      fill(30, 30, 255);
      specularMaterial(255);
      shininess(20);

      // 绘制左侧金属感较低的球体。
      translate(-25, 0, 0);
      metalness(1);
      sphere(20);

      // 绘制右侧金属感较高的球体。
      translate(50, 0, 0);
      metalness(50);
      sphere(20);
    }
    </code>
