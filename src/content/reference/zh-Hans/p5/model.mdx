  <p>绘制一个 a <a href="/reference/p5/p5.Geometry">p5.Geometry</a> 对象到画布上。</p>

  <p>参数 <code>model</code> 是要绘制的

  <a href="/reference/p5/p5.Geometry">p5.Geometry</a> 对象。

  <a href="/reference/p5/p5.Geometry">p5.Geometry</a> 对象可以使用

  <a href="/reference/p5/buildGeometry">buildGeometry()</a>，或

  <a href="/reference/p5/beginGeometry">beginGeometry()</a> 和

  <a href="/reference/p5/endGeometry">endGeometry()</a> 来构建。 它们也可以通过 <a href="/reference/p5/loadGeometry">loadGeometry()</a> 从文件加载。</p>

  <p>注意：<code>model()</code> 只能在 WebGL 模式下使用。</p>

    <code>
    // Click and drag the mouse to view the scene from different angles.

    let shape;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // Create the p5.Geometry object.
      shape = buildGeometry(createShape);

      describe('A white cone drawn on a gray background.');
    }

    function draw() {
      background(200);

      // Enable orbiting with the mouse.
      orbitControl();

      // Draw the p5.Geometry object.
      model(shape);
    }

    // Create p5.Geometry object from a single cone.
    function createShape() {
      cone();
    }

    <code>
    // Click and drag the mouse to view the scene from different angles.

    let shape;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // Create the p5.Geometry object.
      shape = buildGeometry(createArrow);

      describe('Two white arrows drawn on a gray background. The arrow on the right rotates slowly.');
    }

    function draw() {
      background(50);

      // Enable orbiting with the mouse.
      orbitControl();

      // Turn on the lights.
      lights();

      // Style the arrows.
      noStroke();

      // Draw the p5.Geometry object.
      model(shape);

      // Translate and rotate the coordinate system.
      translate(30, 0, 0);
      rotateZ(frameCount * 0.01);

      // Draw the p5.Geometry object again.
      model(shape);
    }

    function createArrow() {
      // Add shapes to the p5.Geometry object.
      push();
      rotateX(PI);
      cone(10);
      translate(0, -10, 0);
      cylinder(3, 20);
      pop();
    }
    </code>
    </div>
