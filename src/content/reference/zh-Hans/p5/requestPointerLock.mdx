description: >
  <p>锁定鼠标指针到其当前位置并使其不可见。</p>

  <p><code>requestPointerLock()</code> 允许鼠标无限移动而不会离开屏幕。调用 <code>requestPointerLock()</code> 会锁定 <a href="/reference/p5/mouseX">mouseX</a>、<a href="/reference/p5/mouseY">mouseY</a>、<a href="/reference/p5/pmouseX">pmouseX</a> 和 <a href="/reference/p5/pmouseY">pmouseY</a> 的值。<a href="/reference/p5/movedX">movedX</a> 和 <a href="/reference/p5/movedY">movedY</a> 会继续更新，并可用于获取自上一帧绘制以来鼠标移动的距离。调用 <a href="/reference/p5/exitPointerLock">exitPointerLock()</a> 会恢复更新鼠标系统变量。</p>

  <p>注意：大多数浏览器在调用 <code>requestPointerLock()</code> 之前需要一个输入操作，例如点击。建议在如 <a href="/reference/p5/doubleClicked">doubleClicked()</a> 等事件函数中调用 <code>requestPointerLock()</code>。</p>

    <div>
    <code>
    let score = 0;

    function setup() {
      createCanvas(100, 100);

      describe(
        'A gray square with the text "Score: X" at its center. The score increases when the user moves the mouse upward. It decreases when the user moves the mouse downward.'
      );
    }

    function draw() {
      background(200);

      // Update the score.
      score -= movedY;

      // Style the text.
      textAlign(CENTER);
      textSize(16);

      // Display the score.
      text(`Score: ${score}`, 50, 50);
    }

    // Lock the pointer when the user double-clicks.
    function doubleClicked() {
      requestPointerLock();
    }
    </code>
