  <p>获取颜色的 alpha（透明度）值。</p>


  <p><code>alpha()</code> 从一个 <a href="/reference/p5/p5.Color">p5.Color</a> 对象、一个颜色组件组成的数组、或


  一个 CSS 颜色字符串中提取 alpha 值。</p>
line: '16'
isConstructor: 'false'
itemtype: method
example:
  - |-

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(200);

      // Create a p5.Color object.
      let c = color(0, 126, 255, 102);

      // Draw the left rectangle.
      noStroke();
      fill(c);
      rect(15, 15, 35, 70);

      // Set 'alphaValue' to 102.
      let alphaValue = alpha(c);

      // Draw the right rectangle.
      fill(alphaValue);
      rect(50, 15, 35, 70);

      describe('Two rectangles. The left one is light blue and the right one is charcoal gray.');
    }
    </code>
    </div>
