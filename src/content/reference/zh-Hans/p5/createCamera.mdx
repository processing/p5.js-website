  <p>创建一个新的 <a href="/reference/p5/p5.Camera">p5.Camera</a> 对象并将之设置为当前的（活跃的）相机。</p>

  <p>新摄像机初始化时的默认位置为 <code>(0, 0, 800)</code>，且默认为透视投影。其属性可通过诸如 <code>myCamera.lookAt(0, 0, 0)</code> 的 <a href="/reference/p5/p5.Camera">p5.Camera</a> 方法控制。</p>

  <p>每个 3D 草图开始时都会初始化一个默认摄像头。可以使用 <a href="/reference/p5/camera">camera()</a>、<a href="/reference/p5/perspective">perspective()</a>、<a href="/reference/p5/ortho">ortho()</a> 和 <a href="/reference/p5/frustum">frustum()</a> 函数进行控制。</p>

  <p>注意：<code>createCamera()</code> 只能在 WebGL 模式下使用。</p>

    <code>
    // Double-click to toggle between cameras.

    let cam1;
    let cam2;
    let usingCam1 = true;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // Create the first camera.
      // Keep its default settings.
      cam1 = createCamera();

      // Create the second camera.
      // Place it at the top-left.
      // Point it at the origin.
      cam2 = createCamera();
      cam2.setPosition(400, -400, 800);
      cam2.lookAt(0, 0, 0);

      // Set the current camera to cam1.
      setCamera(cam1);

      describe('A white cube on a gray background. The camera toggles between frontal and aerial views when the user double-clicks.');
    }

    function draw() {
      background(200);

      // Draw the box.
      box();
    }

    // Toggle the current camera when the user double-clicks.
    function doubleClicked() {
      if (usingCam1 === true) {
        setCamera(cam2);
        usingCam1 = false;
      } else {
        setCamera(cam1);
        usingCam1 = true;
      }
    }
