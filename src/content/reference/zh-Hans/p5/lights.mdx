    <code>
    // 单击并拖动鼠标以从不同角度查看场景。
    // 双击以打开灯光。

    let isLit = false;

    function setup() {
      createCanvas(100, 100, WEBGL);

      describe('一个白色盒子绘制在灰色背景上。当用户双击时，光线的质量会发生变化。');
    }

    function draw() {
      background(50);

      // 使用鼠标进行轨道控制。
      orbitControl();

      // 控制灯光。
      if (isLit === true) {
        lights();
      }

      // 绘制盒子。
      box();
    }

    // 当用户双击时打开灯光。
    function doubleClicked() {
      isLit = true;
    }
    </code>
