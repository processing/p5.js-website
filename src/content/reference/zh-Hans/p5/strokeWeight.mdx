  <p>设置点、线和形状轮廓的笔画宽度。</p>


  <p>需要注意的是：<code>strokeWeight()</code> 会受到形状变换的影响，特别是在使用了 <a href="./scale">scale()</a> 函数的情况下。</p>
line: '540'
isConstructor: 'false'
itemtype: method
example:
  - |-

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(200);

      // Top.
      line(20, 20, 80, 20);

      // Middle.
      strokeWeight(4);
      line(20, 40, 80, 40);

      // Bottom.
      strokeWeight(10);
      line(20, 70, 80, 70);

      describe('Three horizontal black lines. The top line is thin, the middle is medium, and the bottom is thick.');
    }
    </code>
    </div>

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(200);

      // Top.
      line(20, 20, 80, 20);

      // Scale by a factor of 5.
      scale(5);

      // Bottom. Coordinates are adjusted for scaling.
      line(4, 8, 16, 8);

      describe('Two horizontal black lines. The top line is thin and the bottom is five times thicker than the top.');
    }
    </code>
    </div>
