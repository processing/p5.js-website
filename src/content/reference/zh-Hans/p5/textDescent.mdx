title: textDescent
module: 字体
submodule: 属性
file: src/typography/attributes.js
description: |2-

  <p>以字体当前大小计算当前字体的下降。</p>

  <p>下降代表最长下降字符的距离，以像素为单位，低于基线。</p>
line: '396'
isConstructor: 'false'
itemtype: method
example:
  - |-

    <div>
    <code>
    let font;

    function preload()  {
      font = loadFont('/assets/inconsolata.otf');
    }

    function setup()  {
      createCanvas(100, 100);

      background(200);

      // Style the font.
      textFont(font);

      // Different for each font.
      let fontScale = 0.9;

      let baseY = 75;
      strokeWeight(0.5);

      // Draw small text.
      textSize(24);
      text('dp', 0, baseY);

      // Draw baseline and descent.
      let d = textDescent() * fontScale;
      line(0, baseY, 23, baseY);
      line(23, baseY, 23, baseY + d);

      // Draw large text.
      textSize(48);
      text('dp', 45, baseY);

      // Draw baseline and descent.
      d = textDescent() * fontScale;
      line(45, baseY, 91, baseY);
      line(91, baseY, 91, baseY + d);

      describe('The letters "dp" written twice in different sizes. Each version has a horizontal baseline. A vertical line extends downward from each baseline to the bottom of the "p".');
    }
    </code>
    </div>
class: p5
return:
  description: 以像素为单位测量的下降值。
