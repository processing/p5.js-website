      let c = color(0, 50, 100);

      // Draw the left rectangle.
      noStroke();
      fill(c);
      rect(15, 20, 35, 60);

      // Set 'hueValue' to 0.
      let hueValue = hue(c);

      // Draw the right rectangle.
      fill(hueValue);
