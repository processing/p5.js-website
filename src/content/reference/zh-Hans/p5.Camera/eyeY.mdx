      // Style the text.

      textSize(16);
      textFont(font);
      fill(0);

      // Calculate the new y-coordinate.
      let y = 25 * sin(frameCount * 0.01) - 400;

      cam.setPosition(0, y, 800);

      text(`eyeY: ${round(cam.eyeY)}`, 0, 55);
