  - |-

    function setup() {
      background(200);

      textSize(28);
      strokeWeight(0.5);
      let s = 'yoyo';
      let w = textWidth(s);
      text(s, 22, 55);
      line(22, 55, 22 + w, 55);

      describe('The word "yoyo" underlined.');
    }
    </code>

      let w = textWidth(s);
      text(s, 22, 55);
      line(22, 55, 22 + w, 55);

      describe('The word "yo" written twice, one copy beneath the other. The words are divided by a horizontal line.');
    }
    </code>
