  <p>터치스크린 장치에서 현재 터치 포인트들을 저장하는 <code>Array</code>입니다.</p>

  <p><code>touches</code> 배열은 기본적으로 비어 있습니다. 사용자가 화면을 터치하면 새로운 터치 포인트가 추적되어 배열에 추가됩니다. 
  터치 포인트는 다음과 같은 속성을 가진 <code>Object</code>입니다:</p>
  
  <code>

  // 터치 배열을 순회합니다.

  for (let touch of touches) {
    // 캔버스의 왼쪽 상단을 기준으로 한 x 좌표.
    console.log(touch.x);

  <p>  // 캔버스의 왼쪽 상단을 기준으로 한 y 좌표.
    console.log(touch.y);</p>
  <p>  // 브라우저의 왼쪽 상단을 기준으로 한 x 좌표.
    console.log(touch.winX);</p>
  <p>  // 브라우저의 왼쪽 상단을 기준으로 한 y 좌표.
    console.log(touch.winY);</p>
  <p>  // ID 숫자.
    console.log(touch.id);
  }

  </p></code>
