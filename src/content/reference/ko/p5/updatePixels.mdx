  <p><a href="/reference/p5/pixels">pixels</a> 배열에 있는 RGBA 값으로 캔버스를 업데이트합니다.</p>
  
  <p><code>updatePixels()</code>는 <a href="/reference/p5/pixels">pixels</a> 배열의 값을 변경한 후에만 호출합니다.
  
  이러한 변경은 <a href="/reference/p5/loadPixels">loadPixels()</a>를 호출한 직후에 직접 변경하거나
  
  <a href="/reference/p5/set">set()</a>을 호출하여 할 수 있습니다.</p>

    <div>
    <code>
    let img;

    // Load the image.
    function preload() {
      img = loadImage('/assets/rockies.jpg');
    }

    function setup() {
      createCanvas(100, 100);

      // Display the image.
      image(img, 0, 0, 100, 100);

      // Get the pixel density.
      let d = pixelDensity();

      // Calculate the halfway index in the pixels array.
      let halfImage = 4 * (d * width) * (d * height / 2);

      // Load the pixels array.
      loadPixels();

      // Copy the top half of the canvas to the bottom.
      for (let i = 0; i < halfImage; i += 1) {
        pixels[i + halfImage] = pixels[i];
      }

      // Update the canvas.
      updatePixels();

      describe('Two identical images of mountain landscapes, one on top of the other.');
    }
