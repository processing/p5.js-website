    function setup() {
      createCanvas(100, 100);

      // Create an h5 element with the content "p5*js".

      // Set the element's style and position.
      h5.style('color', 'deeppink');
      h5.position(30, 15);

      describe('The text "p5*js" written in pink in the middle of a gray square.');
    }
