    function setup() {

      // Get random coordinates.
      let x = random(0, 100);
      let y = random(0, 100);

      // Draw the white circle.
      circle(x, y, 10);

      // Set a random seed for consistency.
      randomSeed(99);

      // Get random coordinates.
      x = random(0, 100);
      y = random(0, 100);

      // Draw the black circle.
      fill(0);
      circle(x, y, 10);
