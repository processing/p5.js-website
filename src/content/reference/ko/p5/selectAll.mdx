  <p>페이지에서 주어진<a
  href="https://developer.mozilla.org/en-US/docs/Learn/Getting_started_with_the_web/CSS_basics#different_types_of_selectors"
  target="_blank">CSS 선택자 문자열</a>과 일치하는 모든 요소를 검색합니다.</p>


  <p>선택자 문자열은 ID, 클래스, 태그명, 또는 그 조합일 수 있습니다.


  <code>selectAll()</code>은 일치하는 값을 찾으면 해당하는 <a
  href="./p5.Element">p5.Element</a> 객체의 배열을 반환하고, 찾지 못하면 빈 배열을 반환합니다.</p>


  <p>두 번째 매개변수 <code>container</code>는 선택적(optional)입니다. 검색할 컨테이너를 지정합니다. <code>container</code>는 CSS 선택자 문자열, <a href="./p5.Element">p5.Element</a> 객체, 또는 <a href="https://developer.mozilla.org/en-US/docs/Web/API/HTMLElement"
  target="_blank">HTMLElement</a> 객체가 될 수 있습니다.</p>

    function setup() {
      // Create three buttons and position them.
      let b1 = createButton('1');
      b1.position(0, 0);
      let b2 = createButton('2');
      b2.position(0, 30);
      let b3 = createButton('3');
      b3.position(0, 60);

      // Add a class attribute to each button.
      b1.class('btn');
      b2.class('btn btn-pink');
      b3.class('btn');

      // Select the buttons by their class.
      let buttons = selectAll('.btn');
      let pinkButtons = selectAll('.btn-pink');

      // Style the selected buttons.
      buttons.forEach(setFont);
      pinkButtons.forEach(setColor);

      describe('Three buttons stacked vertically. The buttons are labeled, "1", "2", and "3". Buttons "1" and "3" are gray. Button "2" is pink.');
    }

    // Set a button's font to Comic Sans MS.
    function setFont(btn) {
      btn.style('font-family', 'Comic Sans MS');
    }

    // Set a button's background and font color.
    function setColor(btn) {
      btn.style('background', 'deeppink');
      btn.style('color', 'white');
    }
    </code>
