  <a href="/reference/p5/ambientLight">ambientLight()</a>,

  <a href="/reference/p5/directionalLight">directionalLight()</a>,

  <a href="/reference/p5/pointLight">pointLight()</a>,

  <a href="/reference/p5/spotLight">spotLight()</a> 등으로 생성된 모든 빛을 제거합니다. 이 함수들은 <code>noLights()</code> 이후에 호출되어 새로운 조명 구성을 만들 수 있습니다.</p>

    <code>
    // Click and drag the mouse to view the scene from different angles.

    function setup() {
      createCanvas(100, 100, WEBGL);

      describe('Two spheres drawn against a gray background. The top sphere is white and the bottom sphere is red.');
    }

    function draw() {
      background(50);

      // Enable orbiting with the mouse.
      orbitControl();

      // Turn on the lights.
      lights();

      // Style the spheres.
      noStroke();

      // Draw the top sphere.
      push();
      translate(0, -25, 0);
      sphere(20);
      pop();

      // Turn off the lights.
      noLights();

      // Add a red directional light that points into the screen.
      directionalLight(255, 0, 0, 0, 0, -1);

      // Draw the bottom sphere.
      push();
      translate(0, 25, 0);
      sphere(20);
      pop();
    }
