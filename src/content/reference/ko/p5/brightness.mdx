  <p>기본적으로, <code>brightness()</code> 는 다음과 같이 색상의 HSB 밝기값을 반환합니다
  0의 범위


  ~100까지. <a href="/reference/colorMode">colorMode()</a> 가 HSB로 설정된 경우,
  그것을


  지정된 범위의 밝기 값을 반환합니다.</p>

    function setup() {

      // Use HSB color.
      colorMode(HSB);

      let c = color(0, 50, 100);

      // Draw the left rectangle.
      noStroke();
      fill(c);
      rect(15, 15, 35, 70);

      // Set 'brightValue' to 100.
      let brightValue = brightness(c);

      // Draw the right rectangle.
      fill(brightValue);
      rect(50, 15, 35, 70);
