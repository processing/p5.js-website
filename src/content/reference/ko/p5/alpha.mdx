      let c = color(0, 126, 255, 102);

      // Draw the left rectangle.
      noStroke();
      fill(c);
      rect(15, 15, 35, 70);

      // Set 'alphaValue' to 102.
      let alphaValue = alpha(c);

      // Draw the right rectangle.
      fill(alphaValue);
      rect(50, 15, 35, 70);
