  <p>A class to describe an

    function setup() {
      createCanvas(100, 100);

      background(200);

      // Create a button element and
      // place it beneath the canvas.
      let btn = createButton('change');
      btn.position(0, 100);

      // Call randomColor() when

      btn.mousePressed(randomColor);
