      // Get random coordinates.

      let y = random(0, 100);

      // Draw the white circle.
      circle(x, y, 10);

      // Set a random seed for consistency.
      randomSeed(99);

      // Get random coordinates.
      x = random(0, 100);
      y = random(0, 100);

      fill(0);
