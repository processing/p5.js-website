  <p>किसी रंग का रंग मान प्राप्त होता है।</p>

  <p><code>hue()</code> <a href="/reference/p5/p5.Color">p5.Color</a> ऑब्जेक्ट, रंग घटकों की एक सरणी, या से रंग मान निकालता है एक सीएसएस रंग स्ट्रिंग.</p>

  <p>ह्यू रंग चक्र पर रंग की स्थिति का वर्णन करता है। डिफ़ॉल्ट रूप से, <code>hue()</code> 0 से 360 की रेंज में रंग का HSL रंग लौटाता है। यदि <a href="/reference/colorMode">colorMode()</a> HSB पर सेट है या एचएसएल, यह दिए गए मोड में ह्यू मान लौटाता है।</p>

    function setup() {
      createCanvas(100, 100);

      background(200);

      // Use HSL color.
      colorMode(HSL);

      // Create a p5.Color object.
      let c = color(0, 50, 100);

      // Draw the left rectangle.
      noStroke();
      fill(c);
      rect(15, 20, 35, 60);

      // Set 'hueValue' to 0.
      let hueValue = hue(c);

      // Draw the right rectangle.
      fill(hueValue);
      rect(50, 20, 35, 60);

      describe(

    }
