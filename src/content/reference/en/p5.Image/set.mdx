description: >
  <p>Sets the color of one or more pixels within an image.</p>

  <p><code>img.set()</code> is easy to use but it's not as fast as

  <a href="/reference/p5.Image/pixels/">img.pixels</a>. Use

  <a href="/reference/p5.Image/pixels/">img.pixels</a> to set many pixel
  values.</p>

  <p><code>img.set()</code> interprets the first two parameters as x- and
  y-coordinates. It

  interprets the last parameter as a grayscale value, a <code>[R, G, B,
  A]</code> pixel

  array, a <a href="/reference/p5/p5.Color">p5.Color</a> object, or another

  <a href="/reference/p5/p5.Image">p5.Image</a> object.</p>

  <p><a href="/reference/p5.Image/updatePixels/">img.updatePixels()</a> must be
  called

  after using <code>img.set()</code> for changes to appear.</p>

      createCanvas(100, 100);

      describe('Four black dots arranged in a square drawn on a gray background.');

    function setup() {
      createCanvas(100, 100);

      background(200);

      // Create a p5.Image object.
      let img = createImage(66, 66);

      // Draw a color gradient.
      for (let x = 0; x < img.width; x += 1) {
        for (let y = 0; y < img.height; y += 1) {
          let c = map(x, 0, img.width, 0, 255);
          img.set(x, y, c);
        }
      }

      // Update the image.
      img.updatePixels();

      // Display the image.
      image(img, 17, 17);

      describe('A square with a horiztonal color gradient from black to white drawn on a gray background.');
    }
