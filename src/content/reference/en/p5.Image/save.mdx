description: >
  <p>Saves the image to a file.</p>

  <p>By default, <code>img.save()</code> saves the image as a PNG image called

  <code>untitled.png</code>.</p>

  <p>The first parameter, <code>filename</code>, is optional. It's a string that
  sets the

  file's name. If a file extension is included, as in

  <code>img.save('drawing.png')</code>, then the image will be saved using that

  format.</p>

  <p>The second parameter, <code>extension</code>, is also optional. It sets the
  files format.

  Either <code>'png'</code> or <code>'jpg'</code> can be used. For example,
  <code>img.save('drawing', 'jpg')</code>

  saves the canvas to a file called <code>drawing.jpg</code>.</p>

  <p>Note: The browser will either save the file immediately or prompt the user

  with a dialogue window.</p>

  <p>The image will only be downloaded as an animated GIF if it was loaded

  from a GIF file. See <a href="/reference/p5/saveGif/">saveGif()</a> to create
  new

  GIFs.</p>

    let img;

    // Load the image.
    function preload() {
      img = loadImage('/assets/rockies.jpg');

    function setup() {
      createCanvas(100, 100);

      // Display the image.
      image(img, 0, 0);

      describe('An image of a mountain landscape. The image is downloaded when the user presses the "s", "j", or "p" key.');
    }

    // Save the image with different options when the user presses a key.
    function keyPressed() {
      if (key === 's') {
        img.save();
      } else if (key === 'j') {
        img.save('rockies.jpg');
      } else if (key === 'p') {
        img.save('rockies', 'png');
      }
    }
