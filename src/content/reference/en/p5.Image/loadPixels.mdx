      createCanvas(100, 100);

      background(200);

      // Create a p5.Image object.
      let img = createImage(66, 66);

      // Load the image's pixels.
      img.loadPixels();

      // Set the pixels to black.
      for (let x = 0; x < img.width; x += 1) {
        for (let y = 0; y < img.height; y += 1) {
          img.set(x, y, 0);
        }
      }

      // Update the image.
      img.updatePixels();

      // Display the image.
      image(img, 17, 17);

    function setup() {
      createCanvas(100, 100);

      background(200);

      // Create a p5.Image object.
      let img = createImage(66, 66);

      // Load the image's pixels.
      img.loadPixels();

      for (let i = 0; i < img.pixels.length; i += 4) {
        // Red.
        img.pixels[i] = 0;
        // Green.
        img.pixels[i + 1] = 0;
        // Blue.
        img.pixels[i + 2] = 0;
        // Alpha.
        img.pixels[i + 3] = 255;
      }

      // Update the image.
      img.updatePixels();

      // Display the image.
      image(img, 17, 17);

      describe('A black square drawn in the middle of a gray square.');
    }
