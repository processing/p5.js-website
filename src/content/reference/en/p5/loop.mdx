description: >
  <p>Resumes the draw loop after <a href="/reference/p5/noLoop/">noLoop()</a>
  has been

  called.</p>

  <p>By default, <a href="/reference/p5/draw/">draw()</a> tries to run 60 times
  per

  second. Calling <a href="/reference/p5/noLoop/">noLoop()</a> stops

  <a href="/reference/p5/draw/">draw()</a> from repeating. The draw loop can be

  restarted by calling <code>loop()</code>.</p>

  <p>The <a href="/reference/p5/isLooping/">isLooping()</a> function can be used
  to check

  whether a sketch is looping, as in <code>isLooping() === true</code>.</p>

      background(200);

      // Calculate the circle's x-coordinate.
      let x = frameCount;

      circle(x, 50, 20);

    // Resume the draw loop when the user double-clicks.
    function doubleClicked() {
      loop();
