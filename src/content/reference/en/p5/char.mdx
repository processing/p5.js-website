description: >
  <p>Converts a <code>Number</code> or <code>String</code> to a single-character
  <code>String</code>.</p>

  <p><code>char()</code> converts numbers to their single-character string
  representations.</p>

  <p>The parameter, <code>n</code>, is the value to convert. If a number is
  passed, as in

  <code>char(65)</code>, the corresponding single-character string is returned.
  If a

  string is passed, as in <code>char('65')</code>, the string is converted to an
  integer

  (whole number) and the corresponding single-character string is returned.

  If an array is passed, as in <code>char([65, 66, 67])</code>, an array of

  single-character strings is returned.</p>

  <p>See <a
  href="https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/String/fromCharCode/"
  target="_blank">MDN</a>

  for more information about conversions.</p>

    function setup() {
      createCanvas(100, 100);

      background(200);

      // Create an array of numbers.
      let original = ['65', 66, '67'];

      // Convert the string to a char.
      let converted = char(original);

      // Style the text.
      textAlign(CENTER, CENTER);
      textSize(16);

      // Iterate over elements of the converted array.
      for (let i = 0; i < converted.length; i += 1) {

        // Calculate the y-coordinate.
        let y = (i + 1) * 25;

        // Display the original and converted values.
        text(`${original[i]} : ${converted[i]}`, 50, y);
      }

      describe(
        'The text "65 : A", "66 : B", and "67 : C" written on three separate lines. The text is in black on a gray background.'
