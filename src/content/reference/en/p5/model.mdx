description: >
  <p>Draws a <a href="/reference/p5/p5.Geometry">p5.Geometry</a> object to the
  canvas.</p>

  <p>The parameter, <code>model</code>, is the

  <a href="/reference/p5/p5.Geometry">p5.Geometry</a> object to draw.

  <a href="/reference/p5/p5.Geometry">p5.Geometry</a> objects can be built with

  <a href="/reference/p5/buildGeometry/">buildGeometry()</a>, or

  <a href="/reference/p5/beginGeometry/">beginGeometry()</a> and

  <a href="/reference/p5/endGeometry/">endGeometry()</a>. They can also be
  loaded from

  a file with <a href="/reference/p5/loadGeometry/">loadGeometry()</a>.</p>

  <p>Note: <code>model()</code> can only be used in WebGL mode.</p>

    <div>
    <code>
    // Click and drag the mouse to view the scene from different angles.

    let shape;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // Create the p5.Geometry object.
      shape = buildGeometry(createShape);

      describe('A white cone drawn on a gray background.');
    }

    function draw() {
      background(200);

      // Enable orbiting with the mouse.
      orbitControl();

      // Draw the p5.Geometry object.
      model(shape);
    }

    // Create p5.Geometry object from a single cone.
    function createShape() {
      cone();
    }

    <code>
    // Click and drag the mouse to view the scene from different angles.

    let shape;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // Create the p5.Geometry object.
      shape = buildGeometry(createArrow);

      describe('Two white arrows drawn on a gray background. The arrow on the right rotates slowly.');
    }

    function draw() {
      background(50);

      // Enable orbiting with the mouse.
      orbitControl();

      // Turn on the lights.
      lights();

      // Style the arrows.
      noStroke();

      // Draw the p5.Geometry object.
      model(shape);

      // Translate and rotate the coordinate system.
      translate(30, 0, 0);
      rotateZ(frameCount * 0.01);

      // Draw the p5.Geometry object again.
      model(shape);
    }

    function createArrow() {
      // Add shapes to the p5.Geometry object.
      push();
      rotateX(PI);
      cone(10);
      translate(0, -10, 0);
      cylinder(3, 20);
      pop();
    }
    </code>
    </div>
