description: >-
  <p>A shader hook block that sets the color for each pixel in a filter shader.
  This hook can be used inside <a
  href="/reference/p5/buildFilterShader/"><code>buildFilterShader()</code></a>
  to control the output color for each pixel.</p>

  <p><code>filterColor</code> has the following properties:</p>

  <ul><li><code>texCoord</code>: a two-component vector representing the texture
  coordinates (u, v).</li><li><code>canvasSize</code>: a two-component vector
  representing the canvas size in pixels (width,
  height).</li><li><code>texelSize</code>: a two-component vector representing
  the size of a single texel in texture
  space.</li><li><code>canvasContent</code>: a texture containing the sketch's
  contents before the filter is applied.</li></ul><p>Call <code>.set()</code> on
  the hook with a vector with four components (red, green, blue, alpha) to
  update the final color.</p>

  <p>This hook is available in:</p>

  <ul><li><a
  href="/reference/p5/buildFilterShader/"><code>buildFilterShader()</code></a></li></ul>

  - |-
    let myShader;
    function setup() {
      createCanvas(200, 200, WEBGL);
      myShader = buildFilterShader(warp);
    }

    function warp() {
      filterColor.begin();
      // Warp the texture coordinates for a wavy effect
      let warped = [
        filterColor.texCoord.x,
        filterColor.texCoord.y + 0.1 * sin(filterColor.texCoord.x * 10)
      ];
      filterColor.set(getTexture(canvasContent, warped));
      filterColor.end();
    }

    function draw() {
      background(180);
      // Draw something to the canvas
      fill('yellow');
      circle(0, 0, 150);
      filter(myShader);
    }
