description: >
  <p>Returns the number of milliseconds since a sketch started running.</p>

  <p><code>millis()</code> keeps track of how long a sketch has been running in

  milliseconds (thousandths of a second). This information is often

  helpful for timing events and animations.</p>

  <p>If a sketch has a

  <a href="/reference/p5/setup/">setup()</a> function, then
  <code>millis()</code> begins tracking

  time before the code in <a href="/reference/p5/setup/">setup()</a> runs. If a

  sketch includes a <a href="/reference/p5/preload/">preload()</a> function,
  then

  <code>millis()</code> begins tracking time as soon as the code in

  <a href="/reference/p5/preload/">preload()</a> starts running.</p>

    function setup() {

      let ms = millis();

      // Style the text.
      textAlign(LEFT, CENTER);
      textSize(10);
      textFont('Courier New');

      // Display how long it took setup() to be called.
      text(`Startup time: ${round(ms, 2)} ms`, 5, 50, 90);
