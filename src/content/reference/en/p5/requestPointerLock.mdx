  <p>Locks the mouse pointer to its current position and makes it invisible.</p>

  <p><code>requestPointerLock()</code> allows the mouse to move forever without
  leaving the

  screen. Calling <code>requestPointerLock()</code> locks the values of

  <a href="/reference/p5/mouseX/">mouseX</a>, <a
  href="/reference/p5/mouseY/">mouseY</a>,

  <a href="/reference/p5/pmouseX/">pmouseX</a>, and <a
  href="/reference/p5/pmouseY/">pmouseY</a>.

  <a href="/reference/p5/movedX/">movedX</a> and <a
  href="/reference/p5/movedY/">movedY</a>

  continue updating and can be used to get the distance the mouse moved since

  the last frame was drawn. Calling

  <a href="/reference/p5/exitPointerLock/">exitPointerLock()</a> resumes
  updating the

  mouse system variables.</p>

  <p>Note: Most browsers require an input, such as a click, before calling

  <code>requestPointerLock()</code>. Itâ€™s recommended to call
  <code>requestPointerLock()</code> in

  an event function such as <a
  href="/reference/p5/doubleClicked/">doubleClicked()</a>.</p>

    let score = 0;

    function setup() {
      createCanvas(100, 100);

      describe(
        'A gray square with the text "Score: X" at its center. The score increases when the user moves the mouse upward. It decreases when the user moves the mouse downward.'
      );
    }

    function draw() {
      background(200);

      // Update the score.
      score -= movedY;

      // Style the text.
      textAlign(CENTER);
      textSize(16);

      // Display the score.
      text(`Score: ${score}`, 50, 50);
    }

    // Lock the pointer when the user double-clicks.
    function doubleClicked() {
      requestPointerLock();
    }
