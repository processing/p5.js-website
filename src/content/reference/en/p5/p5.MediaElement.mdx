  <p>A class to handle audio and video.</p>

  <p><code>p5.MediaElement</code> extends <a href="./p5.Element">p5.Element</a>
  with

  methods to handle audio and video. <code>p5.MediaElement</code> objects are
  created by

  calling <a href="./createVideo">createVideo</a>,

  <a href="./createAudio">createAudio</a>, and

  <a href="./createCapture">createCapture</a>.</p>

  - |-

    <div class='notest'>
    <code>
    let capture;

    function setup() {
      createCanvas(100, 100);

      // Create a p5.MediaElement using createCapture().
      capture = createCapture(VIDEO);
      capture.hide();

      describe('A webcam feed with inverted colors.');
    }

    function draw() {
      // Display the video stream and invert the colors.
      image(capture, 0, 0, width, width * capture.height / capture.width);
      filter(INVERT);

methods:
  play:
    description: |
      <p>Plays audio or video from a media element.</p>
    path: p5.MediaElement/play
  stop:
    description: >
      <p>Stops a media element and sets its current time to 0.</p>

      <p>Calling <code>media.play()</code> will restart playing audio/video from
      the beginning.</p>
    path: p5.MediaElement/stop
  pause:
    description: >
      <p>Pauses a media element.</p>

      <p>Calling <code>media.play()</code> will resume playing audio/video from
      the moment it paused.</p>
    path: p5.MediaElement/pause
  loop:
    description: |
      <p>Plays the audio/video repeatedly in a loop.</p>
    path: p5.MediaElement/loop
  noLoop:
    description: |
      <p>Stops the audio/video from playing in a loop.</p>
      <p>The media will stop when it finishes playing.</p>
    path: p5.MediaElement/noLoop
  autoplay:
    description: >
      <p>Sets the audio/video to play once it's loaded.</p>

      <p>The parameter, <code>shouldAutoplay</code>, is optional. Calling

      <code>media.autoplay()</code> without an argument causes the media to play

      automatically. If <code>true</code> is passed, as in
      <code>media.autoplay(true)</code>, the

      media will automatically play. If <code>false</code> is passed, as in

      <code>media.autoPlay(false)</code>, it won't play automatically.</p>
    path: p5.MediaElement/autoplay
  volume:
    description: >
      <p>Sets the audio/video volume.</p>

      <p>Calling <code>media.volume()</code> without an argument returns the
      current volume

      as a number in the range 0 (off) to 1 (maximum).</p>

      <p>The parameter, <code>val</code>, is optional. It's a number that sets
      the volume

      from 0 (off) to 1 (maximum). For example, calling
      <code>media.volume(0.5)</code>

      sets the volume to half of its maximum.</p>
    path: p5.MediaElement/volume
  speed:
    description: >
      <p>Sets the audio/video playback speed.</p>

      <p>The parameter, <code>val</code>, is optional. It's a number that sets
      the playback

      speed. 1 plays the media at normal speed, 0.5 plays it at half speed, 2

      plays it at double speed, and so on. -1 plays the media at normal speed

      in reverse.</p>

      <p>Calling <code>media.speed()</code> returns the current speed as a
      number.</p>

      <p>Note: Not all browsers support backward playback. Even if they do,

      playback might not be smooth.</p>
    path: p5.MediaElement/speed
  time:
    description: >
      <p>Sets the media element's playback time.</p>

      <p>The parameter, <code>time</code>, is optional. It's a number that
      specifies the

      time, in seconds, to jump to when playback begins.</p>

      <p>Calling <code>media.time()</code> without an argument returns the
      number of seconds

      the audio/video has played.</p>

      <p>Note: Time resets to 0 when looping media restarts.</p>
    path: p5.MediaElement/time
  duration:
    description: |
      <p>Returns the audio/video's duration in seconds.</p>
    path: p5.MediaElement/duration
  onended:
    description: >
      <p>Calls a function when the audio/video reaches the end of its
      playback.</p>

      <p>The element is passed as an argument to the callback function.</p>

      <p>Note: The function won't be called if the media is looping.</p>
    path: p5.MediaElement/onended
  connect:
    description: >
      <p>Sends the element's audio to an output.</p>

      <p>The parameter, <code>audioNode</code>, can be an <code>AudioNode</code>
      or an object from the

      <code>p5.sound</code> library.</p>

      <p>If no element is provided, as in <code>myElement.connect()</code>, the
      element

      connects to the main output. All connections are removed by the

      <code>.disconnect()</code> method.</p>

      <p>Note: This method is meant to be used with the p5.sound.js addon
      library.</p>
    path: p5.MediaElement/connect
  disconnect:
    description: >
      <p>Disconnect all Web Audio routing, including to the main output.</p>

      <p>This is useful if you want to re-route the output through audio
      effects,

      for example.</p>
    path: p5.MediaElement/disconnect
  showControls:
    description: >
      <p>Show the default

      <a
      href="https://developer.mozilla.org/en-US/docs/Web/API/HTMLMediaElement"
      target="_blank">HTMLMediaElement</a>

      controls.</p>

      <p>Note: The controls vary between web browsers.</p>
    path: p5.MediaElement/showControls
  hideControls:
    description: >
      <p>Hide the default

      <a
      href="https://developer.mozilla.org/en-US/docs/Web/API/HTMLMediaElement"
      target="_blank">HTMLMediaElement</a>

      controls.</p>
    path: p5.MediaElement/hideControls
  addCue:
    description: >
      <p>Schedules a function to call when the audio/video reaches a specific
      time

      during its playback.</p>

      <p>The first parameter, <code>time</code>, is the time, in seconds, when
      the function

      should run. This value is passed to <code>callback</code> as its first
      argument.</p>

      <p>The second parameter, <code>callback</code>, is the function to call at
      the specified

      cue time.</p>

      <p>The third parameter, <code>value</code>, is optional and can be any
      type of value.

      <code>value</code> is passed to <code>callback</code>.</p>

      <p>Calling <code>media.addCue()</code> returns an ID as a string. This is
      useful for

      removing the cue later.</p>
    path: p5.MediaElement/addCue
  removeCue:
    description: |
      <p>Removes a callback based on its ID.</p>
    path: p5.MediaElement/removeCue
  clearCues:
    description: |
      <p>Removes all functions scheduled with <code>media.addCue()</code>.</p>
    path: p5.MediaElement/clearCues
