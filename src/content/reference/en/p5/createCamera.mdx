description: >
  <p>Creates a new <a href="/reference/p5.Camera">p5.Camera</a> object and sets
  it

  as the current (active) camera.</p>

  <p>The new camera is initialized with a default position

  (see <a href="/reference/p5.Camera/camera">camera()</a>)

  and a default perspective projection

  (see <a href="/reference/p5.Camera/perspective">perspective()</a>).

  Its properties can be controlled with the <a
  href="/reference/p5.Camera">p5.Camera</a>

  methods.</p>

  <p>Note: Every 3D sketch starts with a default camera initialized.

  This camera can be controlled with the global methods

  <a href="/reference/p5/camera">camera()</a>,

  <a href="/reference/p5/perspective">perspective()</a>, <a
  href="/reference/p5/ortho">ortho()</a>,

  and <a href="/reference/p5/frustum">frustum()</a> if it is the only camera

  in the scene.</p>

    <div><code>
    // Creates a camera object and animates it around a box.
    let camera;
    function setup() {
      createCanvas(100, 100, WEBGL);
      background(0);
      camera = createCamera();
      camera.camera(0, 0, 50*sqrt(3), 0, 0, 0, 0, 1, 0);
      camera.perspective(PI/3, 1, 5*sqrt(3), 500*sqrt(3));
      describe('An example that creates a camera and moves it around the box.');
    }

    function draw() {
      background(0);
      // The camera will automatically
      // rotate to look at [0, 0, 0].
      camera.lookAt(0, 0, 0);

      // The camera will move on the
      // x axis.
      camera.setPosition(sin(frameCount / 60) * 200, 0, 100);
      box(20);

      // A 'ground' box to give the viewer
      // a better idea of where the camera
      // is looking.
      translate(0, 50, 0);
      rotateX(HALF_PI);
      box(150, 150, 20);
    }
