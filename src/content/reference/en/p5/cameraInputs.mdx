description: >-
  <p>A shader hook block that adjusts vertex properties from the perspective of
  the camera. This hook can be used inside <a
  href="/reference/p5/buildMaterialShader/"><code>buildMaterialShader()</code></a>
  and similar shader <a
  href="/reference/p5.Shader/modify/"><code>modify()</code></a> calls to
  customize vertex positions, normals, texture coordinates, and colors before
  rendering. "Camera space" refers to the coordinate system of the 3D scene
  after transformations have been applied, seen relative to the camera.</p>

  <p><code>cameraInputs</code>  has the following properties:</p>

  <ul><li><code>position</code>: a three-component vector representing the
  position after camera transformation.</li><li><code>normal</code>: a
  three-component vector representing the normal after camera
  transformation.</li><li><code>texCoord</code>: a two-component vector
  representing the texture coordinates.</li><li><code>color</code>: a
  four-component vector representing the color of the vertex (red, green, blue,
  alpha).</li></ul><p>This hook is available in:</p>

  <ul><li><a
  href="/reference/p5/buildColorShader/"><code>buildColorShader()</code></a></li><li><a
  href="/reference/p5/buildMaterialShader/"><code>buildMaterialShader()</code></a></li><li><a
  href="/reference/p5/buildNormalShader/"><code>buildNormalShader()</code></a></li><li><a
  href="/reference/p5/buildStrokeShader/"><code>buildStrokeShader()</code></a></li></ul>

example:
  - |-
    let myShader;
    function setup() {
      createCanvas(200, 200, WEBGL);
      myShader = buildMaterialShader(material);
    }

    function material() {
      let t = uniformFloat();
      cameraInputs.begin();
      // Move vertices in camera space based on their x position
      cameraInputs.position.y += 30 * sin(cameraInputs.position.x * 0.05 + t * 0.001);
      // Tint all vertices blue
      cameraInputs.color.b = 1;
      cameraInputs.end();
    }

    function draw() {
      background(200);
      shader(myShader);
      myShader.setUniform('t', millis());
      noStroke();
      fill('red');
      sphere(50);
    }
