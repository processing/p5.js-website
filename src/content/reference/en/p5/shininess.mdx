description: >
  <p>Sets the amount of gloss ("shininess") of a

  <a href="/reference/p5/specularMaterial/">specularMaterial()</a>.</p>

  <p>Shiny materials focus reflected light more than dull materials.

  <code>shininess()</code> affects the way materials reflect light sources
  including

  <a href="/reference/p5/directionalLight/">directionalLight()</a>,

  <a href="/reference/p5/pointLight/">pointLight()</a>,

  and <a href="/reference/p5/spotLight/">spotLight()</a>.</p>

  <p>The parameter, <code>shine</code>, is a number that sets the amount of
  shininess.

  <code>shine</code> must be greater than 1, which is its default value.</p>

    <div>
    <code>
    function setup() {
      createCanvas(100, 100, WEBGL);

      describe(
        'Two red spheres drawn on a gray background. White light reflects from their surfaces as the mouse moves. The right sphere is shinier than the left sphere.'
      );
    }

    function draw() {
      background(200);

      // Turn on a red ambient light.
      ambientLight(255, 0, 0);

      // Get the mouse's coordinates.
      let mx = mouseX - 50;
      let my = mouseY - 50;

      // Turn on a white point light that follows the mouse.
      pointLight(255, 255, 255, mx, my, 50);

      // Style the sphere.
      noStroke();

      // Add a specular material with a grayscale value.
      specularMaterial(255);

      // Draw the left sphere with low shininess.
      translate(-25, 0, 0);
      shininess(10);
      sphere(20);

      // Draw the right sphere with high shininess.
      translate(50, 0, 0);
      shininess(100);
      sphere(20);
    }
    </code>
