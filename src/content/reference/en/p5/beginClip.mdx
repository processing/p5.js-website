description: >
  <p>Starts defining a shape that will mask any shapes drawn afterward.</p>

  <p>Any shapes drawn between <code>beginClip()</code> and

  <a href="#/p5/endClip">endClip()</a> will add to the mask shape. The mask

  will apply to anything drawn after <a href="#/p5/endClip">endClip()</a>.</p>

  <p>The parameter, <code>options</code>, is optional. If an object with an
  <code>invert</code>

  property is passed, as in <code>beginClip({ invert: true })</code>, it will be
  used to

  set the masking mode. <code>{ invert: true }</code> inverts the mask, creating
  holes

  in shapes that are masked. <code>invert</code> is <code>false</code> by
  default.</p>

  <p>Masks can be contained between the

  <a href="#/p5/push">push()</a> and <a href="#/p5/pop">pop()</a> functions.

  Doing so allows unmasked shapes to be drawn after masked shapes.</p>

  <p>Masks can also be defined in a callback function that's passed to

  <a href="#/p5/clip">clip()</a>.</p>

    function setup() {
      createCanvas(100, 100);

      background(200);

      noStroke();

      // Draw a masked shape.
      push();
      // Create a mask.
      beginClip();
      triangle(15, 37, 30, 13, 43, 37);
      circle(45, 45, 7);
      endClip();

      // Draw a backing shape.
      square(5, 5, 45);
      pop();

      // Translate the origin to the center.
      translate(50, 50);

      // Draw an inverted masked shape.
      push();
      // Create an inverted mask.
      beginClip({ invert: true });
      triangle(15, 37, 30, 13, 43, 37);
      circle(45, 45, 7);
      endClip();

      // Draw a backing shape.
      square(5, 5, 45);
      pop();

      describe('In the top left, a white triangle and circle. In the bottom right, a white square with a triangle and circle cut out of it.');
    }
