description: >
  <p>Toggles full-screen mode or returns the current mode.</p>

  <p>Calling <code>fullscreen(true)</code> makes the sketch full-screen. Calling

  <code>fullscreen(false)</code> makes the sketch its original size.</p>

  <p>Calling <code>fullscreen()</code> without an argument returns
  <code>true</code> if the sketch

  is in full-screen mode and <code>false</code> if not.</p>

  <p>Note: Due to browser restrictions, <code>fullscreen()</code> can only be
  called with

  user input such as a mouse press.</p>

    <div>
    <code>
    function setup() {
      background(200);

      describe('A gray canvas that switches between default and full-screen display when clicked.');
    }

    // If the mouse is pressed,
    // toggle full-screen mode.
    function mousePressed() {
      if (mouseX > 0 && mouseX < width && mouseY > 0 && mouseY < height) {
        let fs = fullscreen();
        fullscreen(!fs);
      }
    }
    </code>
