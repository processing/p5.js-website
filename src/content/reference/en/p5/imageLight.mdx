description: >
  <p>Creates an ambient light from an image.</p>

  <p><code>imageLight()</code> simulates a light shining from all directions.
  The effect is

  like placing the sketch at the center of a giant sphere that uses the image

  as its texture. The image's diffuse light will be affected by

  <a href="/reference/p5/fill/">fill()</a> and the specular reflections will be

  affected by <a href="/reference/p5/specularMaterial/">specularMaterial()</a>
  and

  <a href="/reference/p5/shininess/">shininess()</a>.</p>

  <p>The parameter, <code>img</code>, is the <a
  href="/reference/p5/p5.Image">p5.Image</a> object to

  use as the light source.</p>

    // Click and drag the mouse to view the scene from different angles.

    let img;

    // Load an image and create a p5.Image object.
    function preload() {
      img = loadImage('/assets/outdoor_spheremap.jpg');
    }

    function setup() {
      createCanvas(100, 100, WEBGL);

      describe('A sphere floating above a landscape. The surface of the sphere reflects the landscape.');
    }

    function draw() {
      // Enable orbiting with the mouse.
      orbitControl();

      // Draw the image as a panorama (360Ëš background).
      panorama(img);

      // Add a soft ambient light.
      ambientLight(50);

      // Add light from the image.
      imageLight(img);

      // Style the sphere.
      specularMaterial(20);
      shininess(100);
      noStroke();

      // Draw the sphere.
      sphere(30);
    }
