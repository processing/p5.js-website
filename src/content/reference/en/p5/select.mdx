description: >
  <p>Searches the page for the first element that matches the given

  <a
  href="https://developer.mozilla.org/en-US/docs/Learn/Getting_started_with_the_web/CSS_basics#different_types_of_selectors"
  target="_blank">CSS selector string</a>.</p>

  <p>The selector string can be an ID, class, tag name, or a combination.

  <code>select()</code> returns a <a
  href="/reference/p5/p5.Element">p5.Element</a> object if it

  finds a match and <code>null</code> if not.</p>

  <p>The second parameter, <code>container</code>, is optional. It specifies a
  container to

  search within. <code>container</code> can be CSS selector string, a

  <a href="/reference/p5/p5.Element">p5.Element</a> object, or an

  <a href="https://developer.mozilla.org/en-US/docs/Web/API/HTMLElement/"
  target="_blank">HTMLElement</a> object.</p>

example:
  - |-

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);
      background(200);

      // Select the canvas by its tag.
      let cnv = select('canvas');
      cnv.style('border', '5px deeppink dashed');

      describe('A gray square with a dashed pink border.');
    }
    </code>
    </div>

    <div>
    <code>
    function setup() {
      let cnv = createCanvas(100, 100);

      // Add a class attribute to the canvas.
      cnv.class('pinkborder');

      background(200);

      // Select the canvas by its class.
      cnv = select('.pinkborder');

      // Style its border.
      cnv.style('border', '5px deeppink dashed');

      describe('A gray square with a dashed pink border.');
    }
    </code>
    </div>

    <div>
    <code>
    function setup() {
      let cnv = createCanvas(100, 100);

      // Set the canvas' ID.
      cnv.id('mycanvas');

      background(200);

      // Select the canvas by its ID.
      cnv = select('#mycanvas');

      // Style its border.
      cnv.style('border', '5px deeppink dashed');

      describe('A gray square with a dashed pink border.');
    }
    </code>
    </div>
