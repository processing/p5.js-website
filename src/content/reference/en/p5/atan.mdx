description: >
  <p>Calculates the arc tangent of a number.</p>

  <p><code>atan()</code> is the inverse of <a
  href="/reference/p5/tan/">tan()</a>. It expects input

  values in the range of -Infinity to Infinity. By default, <code>atan()</code>
  returns

  values in the range -π ÷ 2 (about -1.57) to π ÷ 2

  (about 1.57). If the <a href="/reference/p5/angleMode/">angleMode()</a> is
  <code>DEGREES</code>

  then values are returned in the range -90 to 90.</p>

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(200);

      // Calculate tan() and atan() values.
      let a = PI / 3;
      let t = tan(a);
      let at = atan(t);

      // Display the values.
      text(`${round(a, 3)}`, 35, 25);
      text(`${round(t, 3)}`, 35, 50);
      text(`${round(at, 3)}`, 35, 75);

      describe('The numbers 1.047, 1.732, and 1.047 written on separate rows.');
    }
