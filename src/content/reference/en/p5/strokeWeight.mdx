title: strokeWeight
module: Shape
submodule: Attributes
file: src/core/shape/attributes.js
description: >
  <p>Sets the width of the stroke used for points, lines, and the outlines of

  shapes.</p>

  <p>Note: <code>strokeWeight()</code> is affected by transformations,
  especially calls to

  <a href="/reference/p5/scale">scale()</a>.</p>
line: 540
isConstructor: false
itemtype: method
example:
  - |-

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(200);

      // Top.
      line(20, 20, 80, 20);

      // Middle.
      strokeWeight(4);
      line(20, 40, 80, 40);

      // Bottom.
      strokeWeight(10);
      line(20, 70, 80, 70);

      describe('Three horizontal black lines. The top line is thin, the middle is medium, and the bottom is thick.');
    }
    </code>
    </div>

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(200);

      // Top.
      line(20, 20, 80, 20);

      // Scale by a factor of 5.
      scale(5);

      // Bottom. Coordinates are adjusted for scaling.
      line(4, 8, 16, 8);

      describe('Two horizontal black lines. The top line is thin and the bottom is five times thicker than the top.');
    }
    </code>
    </div>
class: p5
params:
  - name: weight
    description: |
      <p>the weight of the stroke (in pixels).</p>
    type: Number
