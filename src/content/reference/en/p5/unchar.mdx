  <p>Converts a single-character <code>String</code> to a

  corresponding

  integer (whole number).</p>

  <p>The parameter, <code>n</code>, is the character to convert. For example,

  <code>unchar('A')</code>, returns the number 65. If an array is passed, as in

  <code>unchar(['A', 'B', 'C'])</code>, an array of integers is returned.</p>

    function setup() {
      createCanvas(100, 100);

      background(200);

      // Create an array of characters.
      let original = ['A', 'B', 'C'];

      // Convert the string to a number.
      let converted = unchar(original);

      // Style the text.
      textAlign(CENTER, CENTER);
      textSize(16);

      // Iterate over elements of the converted array.
      for (let i = 0; i < converted.length; i += 1) {

        // Calculate the y-coordinate.
        let y = (i + 1) * 25;

        // Display the original and converted values.
        text(`${original[i]} : ${converted[i]}`, 50, y);
      }

      describe(
        'The text "A : 65", "B : 66", and "C :67" written on three separate lines. The text is in black on a gray background.'
      );
    }
