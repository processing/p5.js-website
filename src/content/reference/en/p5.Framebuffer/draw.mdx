  <p>Draws to the framebuffer by calling a function that contains drawing

  instructions.</p>

  <p>The parameter, <code>callback</code>, is a function with the drawing
  instructions

  for the framebuffer. For example, calling
  <code>myBuffer.draw(myFunction)</code>

  will call a function named <code>myFunction()</code> to draw to the
  framebuffer.

  Doing so has the same effect as the following:</p>

  <pre><code class="language-js">myBuffer.begin();

  myFunction();

  myBuffer.end();

  </code></pre>

    <div>
    <code>
    // Click the canvas to display the framebuffer.

    let myBuffer;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // Create a p5.Framebuffer object.
      myBuffer = createFramebuffer();

      describe('An empty gray canvas. The canvas gets darker and a rotating, multicolor torus appears while the user presses and holds the mouse.');
    }

    function draw() {
      background(200);

      // Draw to the p5.Framebuffer object.
      myBuffer.draw(bagel);

      // Display the p5.Framebuffer object while
      // the user presses the mouse.
      if (mouseIsPressed === true) {
        image(myBuffer, -50, -50);
      }
    }

    // Draw a rotating, multicolor torus.
    function bagel() {
      background(50);
      rotateY(frameCount * 0.01);
      normalMaterial();
      torus(30);
    }
