description: >
  <p>Returns the first matching child element as a new

  <a href="/reference/p5/p5.XML">p5.XML</a> object.</p>

  <p>The parameter, <code>name</code>, is optional. If a string is passed, as in

  <code>myXML.getChild('cat')</code>, then the first child element with the tag

  <code><cat></code> will be returned. If a number is passed, as in

  <code>myXML.getChild(1)</code>, then the child element at that index will be
  returned.</p>

  - |-

    <div>
    <code>
    let myXML;

    // Load the XML and create a p5.XML object.
    function preload() {
      myXML = loadXML('/assets/animals.xml');
    }

    function setup() {
      createCanvas(100, 100);

      background(200);

      // Get the first child element that is a mammal.
      let goat = myXML.getChild('mammal');

      // Style the text.
      textAlign(CENTER, CENTER);
      textFont('Courier New');
      textSize(14);

      // Get the child element's content.
      let content = goat.getContent();

      // Display the child element's content.
      text(content, 50, 50);

      describe('The word "Goat" written in black on a gray background.');
    }
    </code>
