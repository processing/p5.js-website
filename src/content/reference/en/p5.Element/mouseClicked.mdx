  <p>Calls a function when the mouse is pressed and released over the element.
  Calling <code>myElement.mouseReleased(false)</code> disables the function.</p>

      // Create a canvas element and

      background(200);

      // Call randomColor() when a
      // mouse press ends.
      cnv.mouseClicked(randomColor);

      describe('A gray square changes color when the user releases a mouse press.');
    }

    // Paint the background either

      background(c);
