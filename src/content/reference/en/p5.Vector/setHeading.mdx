  <p>Rotates a 2D vector to a specific angle without changing its magnitude.
  By convention, the positive x-axis has an angle of 0. Angles increase in

  <p>If the vector was created with
  <a href="#/p5/createVector">createVector()</a>, <code>setHeading()</code> uses
  the units of the current <a href="#/p5/angleMode">angleMode()</a>.</p>

      push();
      stroke(myColor);
      strokeWeight(3);
      fill(myColor);
      translate(base.x, base.y);
      line(0, 0, vec.x, vec.y);
      rotate(vec.heading());
      let arrowSize = 7;
      translate(vec.mag() - arrowSize, 0);
      triangle(0, arrowSize / 2, 0, -arrowSize / 2, arrowSize, 0);
