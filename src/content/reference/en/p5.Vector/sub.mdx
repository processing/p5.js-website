description: >
  <p>Subtracts from a vector's <code>x</code>, <code>y</code>, and
  <code>z</code> components.</p>

  <p><code>sub()</code> can use separate numbers, as in <code>v.sub(1, 2,
  3)</code>, another

  <a href="#./p5.Vector">p5.Vector</a> object, as in <code>v.sub(v2)</code>, or
  an array

  of numbers, as in <code>v.sub([1, 2, 3])</code>.</p>

  <p>If a value isn't provided for a component, it won't change. For

  example, <code>v.sub(4, 5)</code> adds 4 to <code>v.x</code>, 5 to
  <code>v.y</code>, and 0 to <code>v.z</code>.

  Calling <code>sub()</code> with no arguments, as in <code>v.sub()</code>, has
  no effect.</p>

  <p>The static version of <code>sub()</code>, as in <code>p5.Vector.sub(v2,
  v1)</code>, returns a new

  <a href="#./p5.Vector">p5.Vector</a> object and doesn't change the

  originals.</p>

      createCanvas(100, 100);

      describe('Three arrows drawn on a gray square. A red and a blue arrow extend from the top left. A purple arrow extends from the tip of the red arrow to the tip of the blue arrow.');
    }

    function draw() {
      background(200);

      let origin = createVector(0, 0);

      // Draw the red arrow.
      let v1 = createVector(50, 50);
      drawArrow(origin, v1, 'red');

      // Draw the blue arrow.
      let v2 = createVector(20, 70);
      drawArrow(origin, v2, 'blue');

      // Purple arrow.
      let v3 = p5.Vector.sub(v2, v1);
      drawArrow(v1, v3, 'purple');
    }

    // Draws an arrow between two vectors.
    function drawArrow(base, vec, myColor) {
      push();
      stroke(myColor);
      strokeWeight(3);
      fill(myColor);
      translate(base.x, base.y);
      line(0, 0, vec.x, vec.y);
      rotate(vec.heading());
      let arrowSize = 7;
      translate(vec.mag() - arrowSize, 0);
      triangle(0, arrowSize / 2, 0, -arrowSize / 2, arrowSize, 0);
      pop();
    }
