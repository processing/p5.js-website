description: >
  <p>Calculates new <code>x</code>, <code>y</code>, and <code>z</code>
  components that are proportionally the

  same distance between two vectors.</p>

  <p>The <code>amt</code> parameter is the amount to interpolate between the old
  vector and

  the new vector. 0.0 keeps all components equal to the old vector's, 0.5 is

  halfway between, and 1.0 sets all components equal to the new vector's.</p>

  <p>The static version of <code>lerp()</code>, as in <code>p5.Vector.lerp(v0,
  v1, 0.5)</code>,

  returns a new <a href="/reference/p5/p5.Vector">p5.Vector</a> object and
  doesn't change

  the original.</p>

    function drawArrow(base, vec, myColor) {

      stroke(myColor);
      strokeWeight(3);
      fill(myColor);
      translate(base.x, base.y);
      line(0, 0, vec.x, vec.y);
      rotate(vec.heading());
      let arrowSize = 7;
      translate(vec.mag() - arrowSize, 0);
      triangle(0, arrowSize / 2, 0, -arrowSize / 2, arrowSize, 0);
      pop();
