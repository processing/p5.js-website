description: >
  <p>Gira la cámara hacia la izquierda y hacia la derecha.</p>

  <p>Mover la cámara no cambia su posición. La rotación ocurre en el espacio

  "local" de la cámara.</p>

  <p>El parámetro, <code>angle</code>, es el ángulo que la cámara debe
  rotar. Pasar un ángulo positivo, como
  <code>myCamera.pan(0.001)</code>, gira la cámara hacia la derecha. Pasar un
  ángulo negativo, como
  <code>myCamera.pan(-0.001)</code>, gira la cámara hacia la izquierda.</p>

  <p>Nota: Tenga en cuenta que los ángulos se interpretan según el modo de ángulo

  actual, establecido por <a href="/reference/p5/angleMode">angleMode()</a>.</p>

    <div>
    <code>
    let cam;
    let delta = 0.001;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // Create a p5.Camera object.
      cam = createCamera();

      // Place the camera at the top-center.
      cam.setPosition(0, -400, 800);

      // Point the camera at the origin.
      cam.lookAt(0, 0, 0);

      describe(
        'A white cube on a gray background. The cube goes in and out of view as the camera pans left and right.'
      );
    }

    function draw() {
      background(200);

      // Pan with the camera.
      cam.pan(delta);

      // Switch directions every 120 frames.
      if (frameCount % 120 === 0) {
        delta *= -1;
      }

      // Draw the box.
      box();
    }
