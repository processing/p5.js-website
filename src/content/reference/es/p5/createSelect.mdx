description: >
  <p>Crea un elemento de menú desplegable <code><select></select></code>.</p>

  <p>El parámetro es opcional. Si se pasa <code>true</code>, como en

  <code>let mySelect = createSelect(true)</code>, entonces el menú desplegable admitirá

  selecciones múltiples. Si se pasa un elemento <code><select></select></code> existente,

  como en <code>let mySelect = createSelect(otherSelect)</code>, el 
  elemento existente se

  envolverá en un nuevo objeto <a
  href="/reference/p5/p5.Element">p5.Element</a>
  
  .</p>

  <p>Los menús desplegables extienden la clase <a href="/reference/p5/p5.Element">p5.Element</a>
  con algunos

  métodos útiles para gestionar las opciones:</p>

  <ul>

  <li><code>mySelect.option(name, [value])</code> agrega una opción al menú.
  El primer parámetro, <code>nombre</code>, es una cadena que establece el nombre 
  y valor de la opción. El segundo parámetro, <code>valor</code>, es opcional. Si se 
  proporciona, establece el valor que corresponde a la clave <code>name</code>. Si 
  ya existe una opción con el <code>name</code>, su valor se cambia a 
  <code>value</code>.</li>

  <li><code>mySelect.value()</code> devuelve el valor de la opción actualmente 
  seleccionada.</li>

  <li><code>mySelect.selected()</code> devuelve la opción actualmente 
  seleccionada.</li>

  <li><code>mySelect.selected(option)</code> selecciona la opción dada de forma 
  predeterminada.</li>

  <li><code>mySelect.disable()</code> marca todo el elemento del menú desplegable como 
  deshabilitado.</li>

  <li><code>mySelect.disable(option)</code> marca una opción dada como 
  deshabilitada.</li>

  <li><code>mySelect.enable()</code> marca todo el elemento del menú desplegable como 
  habilitado.</li>

  <li><code>mySelect.enable(option)</code> marca una opción dada como habilitada.</li>

  </ul>

    let mySelect;

    function setup() {
      createCanvas(100, 100);

      // Create a dropdown and allow multiple selections.
      // Place it beneath the canvas.
      mySelect = createSelect(true);
      mySelect.position(0, 100);

      // Add color options.
      mySelect.option('red');
      mySelect.option('green');
      mySelect.option('blue');
      mySelect.option('yellow');

      describe('A gray square with a dropdown menu beneath it. Colorful circles appear when their color is selected.');
    }

    function draw() {
      background(200);

      // Use the selected value(s) to draw circles.
      let colors = mySelect.selected();
      for (let i = 0; i < colors.length; i += 1) {
        // Calculate the x-coordinate.
        let x = 10 + i * 20;

        // Access the color.
        let c = colors[i];

        // Draw the circle.
        fill(c);
        circle(x, 50, 20);
      }
    }
