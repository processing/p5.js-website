description: >
  <p>Obtiene el valor de saturación de un color.</p>

  <p><code>saturation()</code> extrae el valor de saturación de un

  objeto <a href="/reference/p5/p5.Color">p5.Color</a>, un arreglo de componentes de 
  color, o

  una cadena de color CSS.</p>

  <p>La saturación se escala de manera diferente en HSB y HSL. De forma predeterminada,
  <code>saturation()</code>

  devuelve la saturación HSL de un color en el rango 0 a 100. Si el

  <a href="/reference/colorMode">colorMode()</a> está configurado en HSB o HSL, 
  devuelve el

  valor de saturación en el modo dado.</p>

    function setup() {
      createCanvas(100, 100);

      background(50);

      // Use HSB color.
      colorMode(HSB);

      // Create a p5.Color object.
      let c = color(0, 50, 100);

      // Draw the left rectangle.
      noStroke();
      fill(c);
      rect(15, 15, 35, 70);

      // Set 'satValue' to 50.
      let satValue = saturation(c);

      // Draw the right rectangle.
      fill(satValue);
      rect(50, 15, 35, 70);

      describe('Two rectangles. The left one is salmon pink and the right one is dark gray.');
    }
