description: >
  <p>Trazar una curva usando una spline de Catmull-Rom.</p>

  <p>Las curvas spline pueden formar formas y curvas con suaves pendientes. Son como 
  cables que están sujetos a un conjunto de puntos. Las splines se definen por dos 
  puntos de anclaje y dos puntos de control.</p>

  <p>Los dos primeros parámetros, <code>x1</code> y <code>y1</code>, establecen el 
  primer punto de control. Este punto no se dibuja y se puede pensar en él como el 
  punto de inicio de la curva.</p>

  <p>Los siguientes cuatro parámetros, <code>x2</code>, <code>y2</code>, 
  <code>x3</code>, y <code>y3</code>, establecen los dos puntos de anclaje. Los 
  puntos de anclaje son el punto de inicio y el punto final del segmento visible 
  de la curva.</p>

  <p>El séptimo y octavo parámetros, <code>x4</code> y <code>y4</code>, establecen 
  el último punto de control. Este punto no se dibuja y se puede pensar en él como 
  el punto final de la curva.</p>

  <p>Las curvas spline también se pueden dibujar en 3D usando el modo WebGL. La 
  versión 3D de <code>curve()</code> tiene doce argumentos porque cada punto tiene 
  coordenadas x, y y z.</p>

      createCanvas(100, 100);

      point(5, 26);

    <code>
    let x1 = 5;
    let y1 = 26;
    let isChanging = false;

    function setup() {
      createCanvas(100, 100);

      describe(
        'A gray square with a curve drawn in three segments. The curve is a sideways U shape with red segments on top and bottom, and a black segment on the right. The endpoints of all the segments are marked with dots.'
      );
    }

    function draw() {
      background(200);

      // Draw a black spline curve.
      noFill();
      strokeWeight(1);
      stroke(0);
      curve(x1, y1, 73, 24, 73, 61, 15, 65);

      // Draw red spline curves from the anchor points to the control points.
      stroke(255, 0, 0);
      curve(x1, y1, x1, y1, 73, 24, 73, 61);
      curve(73, 24, 73, 61, 15, 65, 15, 65);

      // Draw the anchor points in black.
      strokeWeight(5);
      stroke(0);
      point(73, 24);
      point(73, 61);

      // Draw the control points in red.
      stroke(255, 0, 0);
      point(x1, y1);
      point(15, 65);
    }

    // Start changing the first control point if the user clicks near it.
    function mousePressed() {
      if (dist(mouseX, mouseY, x1, y1) < 20) {
        isChanging = true;
      }
    }

    // Stop changing the first control point when the user releases the mouse.
    function mouseReleased() {
      isChanging = false;
    }

    // Update the first control point while the user drags the mouse.
    function mouseDragged() {
      if (isChanging === true) {
        x1 = mouseX;
        y1 = mouseY;
      }
    }
    </code>
