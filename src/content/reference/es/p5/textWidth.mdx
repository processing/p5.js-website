    function setup() {
      createCanvas(100, 100);

      background(200);

      // Style the text.
      textSize(28);
      strokeWeight(0.5);

      // Calculate the text width.
      let s = 'yoyo';
      let w = textWidth(s);

      // Display the text.
      text(s, 22, 55);

      // Underline the text.
      line(22, 55, 22 + w, 55);

      describe('The word "yoyo" underlined.');
    }
    </code>

    function setup() {
      createCanvas(100, 100);

      background(200);

      // Style the text.
      textSize(28);
      strokeWeight(0.5);

      // Calculate the text width.
      // "\n" starts a new line.
      let s = 'yo\nyo';
      let w = textWidth(s);

      // Display the text.
      text(s, 22, 55);

      // Underline the text.
      line(22, 55, 22 + w, 55);

      describe('The word "yo" written twice, one copy beneath the other. The words are divided by a horizontal line.');
    }
    </code>
