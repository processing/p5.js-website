description: |
  <p>Calcula el descenso de la fuente actual en su tamaño actual.</p>
  <p>El descenso representa la distancia, en píxeles, del carácter con el mayor descendente por debajo de la línea base.</p>
line: 396
isConstructor: false
itemtype: method
example:
  - |-

    <div>
    <code>
    let font;

    function preload()  {
      font = loadFont('/assets/inconsolata.otf');
    }

    function setup()  {
      createCanvas(100, 100);

      background(200);

      // Style the font.
      textFont(font);

      // Different for each font.
      let fontScale = 0.9;

      let baseY = 75;
      strokeWeight(0.5);

      // Draw small text.
      textSize(24);
      text('dp', 0, baseY);

      // Draw baseline and descent.
      let d = textDescent() * fontScale;
      line(0, baseY, 23, baseY);
      line(23, baseY, 23, baseY + d);

      // Draw large text.
      textSize(48);
      text('dp', 45, baseY);

      // Draw baseline and descent.
      d = textDescent() * fontScale;
      line(45, baseY, 91, baseY);
      line(91, baseY, 91, baseY + d);

      describe('The letters "dp" written twice in different sizes. Each version has a horizontal baseline. A vertical line extends downward from each baseline to the bottom of the "p".');
    }
    </code>
    </div>
