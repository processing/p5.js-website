description: >
  <p>Busca en la página el primer elemento que coincida con la

  <a
  href="https://developer.mozilla.org/en-US/docs/Learn/Getting_started_with_the_web/CSS_basics#different_types_of_selectors"
  target="_blank">cadena de selector CSS</a> dada.</p>

  <p>La cadena de selector puede ser un ID, una clase, un nombre de etiqueta o una combinación.

  <code>select()</code> devuelve un objeto <a
  href="/reference/p5/p5.Element">p5.Element</a> si encuentra una 
  
  coincidencia y <code>null</code> si no.</p>

  <p>El segundo parámetro, <code>container</code>, es opcional. Especifica un 
  contenedor para

  buscar dentro de él. <code>container</code> puede ser una cadena de selector CSS, un

  objeto <a href="/reference/p5/p5.Element">p5.Element</a>, o un

  <a href="https://developer.mozilla.org/en-US/docs/Web/API/HTMLElement"
  target="_blank">HTMLElement</a> object.</p>

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);
      background(200);

      // Select the canvas by its tag.
      let cnv = select('canvas');
      cnv.style('border', '5px deeppink dashed');

      describe('A gray square with a dashed pink border.');
    }
    </code>
    </div>

    <div>
    <code>
    function setup() {
      let cnv = createCanvas(100, 100);

      // Add a class attribute to the canvas.
      cnv.class('pinkborder');

      background(200);

      // Select the canvas by its class.
      cnv = select('.pinkborder');

      // Style its border.
      cnv.style('border', '5px deeppink dashed');

      describe('A gray square with a dashed pink border.');
    }
    </code>
    </div>

    <div>
    <code>
    function setup() {
      let cnv = createCanvas(100, 100);

      // Set the canvas' ID.
      cnv.id('mycanvas');

      background(200);

      // Select the canvas by its ID.
      cnv = select('#mycanvas');

      // Style its border.
      cnv.style('border', '5px deeppink dashed');

      describe('A gray square with a dashed pink border.');
    }
    </code>
    </div>
