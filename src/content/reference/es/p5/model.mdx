  <p>Dibuja un objeto <a href="/reference/p5/p5.Geometry">p5.Geometry</a> en el lienzo.</p>

  <p>El parámetro <code>model</code> es el objeto <a href="/reference/p5/p5.Geometry">p5.Geometry</a> a dibujar.</p>

  <p>Los objetos <a href="/reference/p5/p5.Geometry">p5.Geometry</a> se pueden construir con <a href="/reference/p5/buildGeometry">buildGeometry()</a>, o con <a href="/reference/p5/beginGeometry">beginGeometry()</a> y <a href="/reference/p5/endGeometry">endGeometry()</a>. También se pueden cargar desde un archivo con <a href="/reference/p5/loadGeometry">loadGeometry()</a>.</p>

  <p>Nota: <code>model()</code> solo se puede usar en modo WebGL.</p>

    <code>
    // Click and drag the mouse to view the scene from different angles.

    let shape;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // Create the p5.Geometry object.
      shape = buildGeometry(createShape);

      describe('A white cone drawn on a gray background.');
    }

    function draw() {
      background(200);

      // Enable orbiting with the mouse.
      orbitControl();

      // Draw the p5.Geometry object.
      model(shape);
    }

    // Create p5.Geometry object from a single cone.
    function createShape() {
      cone();
    }

    <code>
    // Click and drag the mouse to view the scene from different angles.

    let shape;

    function setup() {
      createCanvas(100, 100, WEBGL);

      // Create the p5.Geometry object.
      shape = buildGeometry(createArrow);

      describe('Two white arrows drawn on a gray background. The arrow on the right rotates slowly.');
    }

    function draw() {
      background(50);

      // Enable orbiting with the mouse.
      orbitControl();

      // Turn on the lights.
      lights();

      // Style the arrows.
      noStroke();

      // Draw the p5.Geometry object.
      model(shape);

      // Translate and rotate the coordinate system.
      translate(30, 0, 0);
      rotateZ(frameCount * 0.01);

      // Draw the p5.Geometry object again.
      model(shape);
    }

    function createArrow() {
      // Add shapes to the p5.Geometry object.
      push();
      rotateX(PI);
      cone(10);
      translate(0, -10, 0);
      cylinder(3, 20);
      pop();
    }
    </code>
    </div>
