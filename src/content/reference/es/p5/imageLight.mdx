description: >
  <p>Crea una luz ambiental a partir de una imagen.</p>

  <p><code>imageLight()</code> simula una luz que brilla desde todas las direcciones.
  El efecto es como colocar el boceto en el centro de una esfera gigante que utiliza la imagen como su textura.
  La luz difusa de la imagen se verá afectada por <a href="/reference/p5/fill">fill()</a> y las reflexiones especulares se verán afectadas por <a href="/reference/p5/specularMaterial">specularMaterial()</a> y <a href="/reference/p5/shininess">shininess()</a>.</p>

  <p>El parámetro <code>img</code> es el objeto <a href="/reference/p5/p5.Image">p5.Image</a> que se utilizará como fuente de luz.</p>

    <div class="notest">
    <code>
    // Click and drag the mouse to view the scene from different angles.

    let img;

    // Load an image and create a p5.Image object.
    function preload() {
      img = loadImage('/assets/outdoor_spheremap.jpg');
    }

    function setup() {
      createCanvas(100, 100, WEBGL);

      describe('A sphere floating above a landscape. The surface of the sphere reflects the landscape.');
    }

    function draw() {
      // Enable orbiting with the mouse.
      orbitControl();

      // Draw the image as a panorama (360˚ background).
      panorama(img);

      // Add a soft ambient light.
      ambientLight(50);

      // Add light from the image.
      imageLight(img);

      // Style the sphere.
      specularMaterial(20);
      shininess(100);
      noStroke();

      // Draw the sphere.
      sphere(30);
    }
    </code>
