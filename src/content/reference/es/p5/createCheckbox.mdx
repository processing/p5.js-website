description: >
  <p>Crea un elemento de casilla de verificación <code><input></input></code>.</p>

  <p>Las casillas de verificación extienden la clase <a href="/reference/p5/p5.Element">p5.Element</a> 
  con un método 
  
  <code>checked()</code>. Llamar a <code>myBox.checked()</code> devuelve 
  <code>true</code> si la casilla está 
  
  marcada y <code>false</code> si no lo está.</p>

  <p>El primer parámetro, <code>label</code>, es opcional. Es una cadena que 
  establece la etiqueta 
  
  a mostrar junto a la casilla de verificación.</p>

  <p>El segundo parámetro, <code>value</code>, también es opcional. Es un booleano 
  que establece el 
  
  valor de la casilla de verificación.</p>

    <div>
    <code>
    let checkbox;

    function setup() {
      createCanvas(100, 100);

      // Create a checkbox and place it beneath the canvas.
      checkbox = createCheckbox();
      checkbox.position(0, 100);

      describe('A black square with a checkbox beneath it. The square turns white when the box is checked.');
    }

    function draw() {
      // Use the checkbox to set the background color.
      if (checkbox.checked()) {
        background(255);
      } else {
        background(0);
      }
    }
    </code>

    }

    function draw() {
      // Use the checkbox to set the background color.
