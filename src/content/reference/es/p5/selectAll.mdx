description: >
  <p>Busca en la página todos los elementos que coinciden con la

  <a
  href="https://developer.mozilla.org/en-US/docs/Learn/Getting_started_with_the_web/CSS_basics#different_types_of_selectors"
  target="_blank">cadena de selector CSS</a> dada.</p>

  <p>La cadena de selector puede ser un ID, una clase, un nombre de etiqueta o una combinación.

  <code>selectAll()</code> devuelve un arreglo de objetos <a
  href="/reference/p5/p5.Element">p5.Element</a>

  si encuentra alguna coincidencia y un arreglo vacío si no encuentra ninguna.</p>

  <p>El segundo parámetro, <code>container</code>, es opcional. Especifica un 
  contenedor para

  buscar dentro de él. <code>container</code> puede ser una cadena de selector CSS, un

  objeto <a href="/reference/p5/p5.Element">p5.Element</a>, o un

  objeto <a href="https://developer.mozilla.org/en-US/docs/Web/API/HTMLElement"
  target="_blank">HTMLElement</a>.</p>

    function setup() {
      // Create three buttons and position them.
      let b1 = createButton('1');
      b1.position(0, 0);
      let b2 = createButton('2');
      b2.position(0, 30);
      let b3 = createButton('3');
      b3.position(0, 60);

      // Add a class attribute to each button.
      b1.class('btn');
      b2.class('btn btn-pink');
      b3.class('btn');

      // Select the buttons by their class.
      let buttons = selectAll('.btn');
      let pinkButtons = selectAll('.btn-pink');

      // Style the selected buttons.
      buttons.forEach(setFont);
      pinkButtons.forEach(setColor);

      describe('Three buttons stacked vertically. The buttons are labeled, "1", "2", and "3". Buttons "1" and "3" are gray. Button "2" is pink.');
    }

    // Set a button's font to Comic Sans MS.
    function setFont(btn) {
      btn.style('font-family', 'Comic Sans MS');
    }

    // Set a button's background and font color.
    function setColor(btn) {
      btn.style('background', 'deeppink');
      btn.style('color', 'white');
    }
    </code>
