description: >
  <p>Una <code>String</code> constante que se utiliza para establecer el

  <a href="/reference/p5/angleMode">angleMode()</a>.</p>

  <p>Por defecto, funciones como <a href="/reference/p5/rotate">rotate()</a>
  y

  <a href="/reference/p5/sin">sin()</a> esperan ángulos medidos en unidades de
  radianes.

  Al llamar a <code>angleMode(RADIANS)</code> se garantiza que los ángulos se
  miden en unidades de

  radianes. Esto puede ser útil si el

  <a href="/reference/p5/angleMode">angleMode()</a> ha sido establecido en

  <a href="/reference/p5/DEGREES">DEGREES</a>.</p>

  <p>Nota: <code>TWO_PI</code> radianes equivale a 360˚.</p>

    <div>
    <code>
    function setup() {
      createCanvas(100, 100);

      background(200);

      // Use degrees.
      angleMode(DEGREES);

      // Draw a red arc from 0˚ to 90˚.
      fill(255, 0, 0);
      arc(50, 50, 80, 80, 0, 90);

      // Use radians.
      angleMode(RADIANS);

      // Draw a blue arc from HALF_PI to PI.
      fill(0, 0, 255);
      arc(50, 50, 80, 80, HALF_PI, PI);

      describe('The bottom half of a circle drawn on a gray background. The bottom-right quarter is red. The bottom-left quarter is blue.');
    }
