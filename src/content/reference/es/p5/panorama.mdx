  <p>Crea un fondo 3D inmersivo.</p>

  <p><code>panorama()</code> transforma imágenes que contienen contenido de 360˚, como
  mapas o

  HDRIs, en fondos 3D inmersivos que rodean un boceto. Explorar el

  espacio requiere cambiar la perspectiva de la cámara con funciones como

  <a href="/reference/p5/orbitControl">orbitControl()</a> o

  <a href="/reference/p5/camera">camera()</a>.</p>

example:
  - |-

    <div class="notest">
    <code>
    // Click and drag the mouse to view the scene from different angles.

    let img;

    // Load an image and create a p5.Image object.
    function preload() {
      img = loadImage('/assets/outdoor_spheremap.jpg');
    }

    function setup() {
      createCanvas(100 ,100 ,WEBGL);

      describe('A sphere floating above a landscape. The surface of the sphere reflects the landscape. The full landscape is viewable in 3D as the user drags the mouse.');
    }

    function draw() {
      // Add the panorama.
      panorama(img);

      // Enable orbiting with the mouse.
      orbitControl();

      // Use the image as a light source.
      imageLight(img);

      // Style the sphere.
      noStroke();
      specularMaterial(50);
      shininess(200);
      metalness(100);

      // Draw the sphere.
      sphere(30);
    }
    </code>
